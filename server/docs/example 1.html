function dateToMDY(date, lang) {
var dd = date.getDate();
var mm = date.getMonth();
var yyyy = date.getFullYear();
if (dd < 10) {
dd = '0' + dd
}
return getDateObject()[lang][mm] + '/' + dd + '/' + yyyy;
};
function formatAMPM(date) {
var hours = date.getHours();
var minutes = date.getMinutes();
var ampm = hours >= 12 ? 'PM' : 'AM';
hours = hours % 12;
hours = hours ? hours : 12; // the hour '0' should be '12'
minutes = minutes < 10 ? '0' + minutes : minutes;
var strTime = hours + ':' + minutes + ' ' + ampm;
return strTime;
};
function getDateObject(){
var dateen = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];

var datefr = ["Jan","Fév","Mar","Avr","Mai","Juin","Juil","Aoû","Sep","Oct","Nov","Déc"];
return {"en":dateen,"es":datees,"pt":datept,"fr":datefr};
}
function dateToMDYshort(date, lang) {
var dd = date.getDate();
var mm = date.getMonth();
var yyyy = date.getFullYear();
if (dd < 10) {
dd = '0' + dd
}
return mm + '/' + dd + '/' + yyyy;
};
